\begin{align*} f(x) &= \sqrt{4x-1}\\ f(x+h) &= \sqrt{4(x+h)-1} = \sqrt{4x+4h-1}\\ \frac{\Delta f}{\Delta x} &= \frac{f(x+h)-f(x)}{h}\\ &= \frac{\sqrt{4x+4h-1} - \sqrt{4x-1}}{h}\\ &= \frac{(\sqrt{4x+4h-1} - \sqrt{4x-1})(\sqrt{4x+4h-1} + \sqrt{4x-1})}{h(\sqrt{4x+4h-1}+\sqrt{4x-1})}\\ &= \frac{(\sqrt{4x+4h-1})^2 - (\sqrt{4x-1})^2}{h(\sqrt{4x+4h-1}+\sqrt{4x-1})}\\ &= \frac{(4x+4h-1) - (4x-1)}{h(\sqrt{4x+4h-1}+\sqrt{4x-1})}\\ &= \frac{4\cancel{h}}{\cancel{h}}(\sqrt{4x+4h-1}+\sqrt{4x-1})}\\ &= \frac{4}{\sqrt{4x+4h-1}+\sqrt{4x-1}}\\ f'(x) &= \lim_{h \to 0} \frac{\Delta f}{\Delta x}\\ &= \lim_{h \to 0} \frac{4}{\sqrt{4x+4h-1} + \sqrt{4x-1}}\\ &= \frac{4}{\sqrt{4x+4(0)-1} + \sqrt{4x-1}}\\ &= \frac{4}{\sqrt{4x-1} + \sqrt{4x-1}}\\ &= \frac{4}{2\sqrt{4x-1}} = \frac{2}{\sqrt{4x-1}} \end{align*}
