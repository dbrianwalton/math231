\begin{align*} \frac{d}{dt}[\left(A(t)\right)^2 + \left(B(t)\right)^2] &= \frac{d}{dt}[\left(A(t)\right)^2] + \frac{d}{dt}[\left(B(t)\right)^2]\\ &= \frac{d}{dA}[A^2] \cdot \frac{d}{dt}[A(t)] + \frac{d}{dB}[B^2] \cdot \frac{d}{dt}[B(t)]\\ &= 2A \cdot \frac{dA}{dt} + 2B \cdot \frac{dB}{dt} \end{align*}
